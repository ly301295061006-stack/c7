const quaco = document.getElementById("quaco");

let particles = [];

// Click Ä‘á»ƒ di chuyá»ƒn quáº£ cÃ²n

function checkTrung(){
  const qc = quaco.getBoundingClientRect();
  const vg = vong.getBoundingClientRect();

  const qcY = qc.top + qc.height/2;
  const vgX = vg.left + vg.width/2;
  const vgY = vg.top + vg.height/2;
  const dist = Math.sqrt((qcX-vgX)**2 + (qcY-vgY)**2);
  
  if(dist < vg.width/2){
    message.textContent = "ðŸŽ‰ TrÃºng vÃ²ng! ChÃºc má»«ng nÄƒm má»›i!";

function rectLeft(){ return document.getElementById("game").getBoundingClientRect().left; }
function rectTop(){ return document.getElementById("game").getBoundingClientRect().top; }

// Firework effect
function firework(x,y){
  for(let i=0;i<40;i++){
    particles.push({
      x: x, y: y,
      dx: Math.random()*4-2,
      dy: Math.random()*-4-1,
      life: 50,
      color: "hsl(" + Math.random()*360 + ",100%,50%)"
    });
  }
  animate();
}

function animate(){
  ctx.clearRect(0,0,canvas.width,canvas.height);
  particles.forEach((p,i)=>{
    ctx.beginPath();
    ctx.arc(p.x,p.y,3,0,Math.PI*2);
    ctx.fillStyle = p.color;
    ctx.fill();
    p.x+=p.dx;
    p.y+=p.dy;
    p.dy+=0.05;
    p.life--;
    if(p.life<=0) particles.splice(i,1);
  });
  if(particles.length>0) requestAnimationFrame(animate);
}
